### 1. Widget概念

Flutter中一切皆Widget，Widget是UI的基本构建块。

```dart
// StatelessWidget - 无状态组件
class MyButton extends StatelessWidget {
  final String text;
  final VoidCallback onPressed;

  const MyButton({
    Key? key,
    required this.text,
    required this.onPressed,
  }) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return ElevatedButton(
      onPressed: onPressed,
      child: Text(text),
    );
  }
}

// StatefulWidget - 有状态组件
class CounterWidget extends StatefulWidget {
  @override
  _CounterWidgetState createState() => _CounterWidgetState();
}

class _CounterWidgetState extends State<CounterWidget> {
  int _counter = 0;

  void _incrementCounter() {
    setState(() {
      _counter++;
    });
  }

  @override
  Widget build(BuildContext context) {
    return Column(
      children: [
        Text('计数: $_counter'),
        ElevatedButton(
          onPressed: _incrementCounter,
          child: Text('增加'),
        ),
      ],
    );
  }
}
```

## 🔗 API集成

### 1. 网络请求配置

#### HTTP服务
```dart
// lib/core/services/http_service.dart
import 'package:http/http.dart' as http;
import 'dart:convert';

class HttpService {
  static const String baseUrl = 'http://localhost:3000/api/v1';

  static Future<Map<String, dynamic>> get(String endpoint) async {
    try {
      final response = await http.get(
        Uri.parse('$baseUrl$endpoint'),
        headers: {'Content-Type': 'application/json'},
      );

      if (response.statusCode == 200) {
        return json.decode(response.body);
      } else {
        throw Exception('请求失败: ${response.statusCode}');
      }
    } catch (e) {
      throw Exception('网络错误: $e');
    }
  }

  static Future<Map<String, dynamic>> post(
    String endpoint,
    Map<String, dynamic> data,
  ) async {
    try {
      final response = await http.post(
        Uri.parse('$baseUrl$endpoint'),
        headers: {'Content-Type': 'application/json'},
        body: json.encode(data),
      );

      if (response.statusCode == 200 || response.statusCode == 201) {
        return json.decode(response.body);
      } else {
        throw Exception('请求失败: ${response.statusCode}');
      }
    } catch (e) {
      throw Exception('网络错误: $e');
    }
  }
}
```

#### Supabase集成
```dart
// lib/core/services/supabase_service.dart
import 'package:supabase_flutter/supabase_flutter.dart';

class SupabaseService {
  static SupabaseService? _instance;
  static SupabaseService get instance => _instance ??= SupabaseService._();

  SupabaseService._();

  SupabaseClient get client => Supabase.instance.client;

  static Future<void> initialize() async {
    await Supabase.initialize(
      url: 'https://tvjcmleckqovnieuexgu.supabase.co',
      anonKey: 'your_anon_key_here',
    );
  }

  // 获取场景数据
  Future<List<Map<String, dynamic>>> getScenarios() async {
    try {
      final response = await client
          .from('scenarios')
          .select('*')
          .eq('is_active', true)
          .order('display_order');
      return List<Map<String, dynamic>>.from(response);
    } catch (e) {
      throw Exception('获取场景失败: $e');
    }
  }

  // 获取器材数据
  Future<List<Map<String, dynamic>>> getEquipment() async {
    try {
      final response = await client
          .from('equipment')
          .select('*')
          .eq('is_active', true)
          .order('category, name');
      return List<Map<String, dynamic>>.from(response);
    } catch (e) {
      throw Exception('获取器材失败: $e');
    }
  }
}
```

#### API服务层
```dart
// lib/core/services/api_service.dart
class ApiService {
  final SupabaseService _supabaseService = SupabaseService.instance;

  // 获取场景列表
  Future<List<Scenario>> getScenarios() async {
    try {
      final data = await _supabaseService.getScenarios();
      return data.map((json) => Scenario.fromJson(json)).toList();
    } catch (e) {
      throw Exception('获取场景失败: $e');
    }
  }

  // 获取推荐锻炼
  Future<List<Exercise>> getRecommendedExercises({
    required String targetMuscle,
    required List<String> equipment,
    required String scenarioId,
    int duration = 60,
  }) async {
    try {
      final response = await HttpService.post('/exercises/recommend', {
        'targetMuscle': targetMuscle,
        'equipment': equipment,
        'scenarioId': scenarioId,
        'duration': duration,
      });

      final exercisesData = response['exercises'] as List;
      return exercisesData.map((json) => Exercise.fromJson(json)).toList();
    } catch (e) {
      throw Exception('获取推荐锻炼失败: $e');
    }
  }
}
```

```dart
    } catch (e) {
      _errorMessage = e.toString();
    } finally {
      _setLoading(false);
    }
  }

  // 选择场景
  void selectScenario(String scenarioId) {
    // 处理场景选择逻辑
    notifyListeners();
  }

  void _setLoading(bool loading) {
    _isLoading = loading;
    notifyListeners();
  }
}
```
